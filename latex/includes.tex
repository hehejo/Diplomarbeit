        \addvspace{2ex}%
        \pagebreak[3]%
        \noindent%
        \makebox[0pt][l]{%

        {\large\textsf{\textbf{#1}}}\hfill#2%

        \nopagebreak%
        \addvspace{1ex}%
}

        \addvspace{0.5ex}%
        \noindent\hspace{1cm}%
        #1\dotfill#2%
        \par%
        \nopagebreak[2]%
}

        \addvspace{0.2ex}%

        \par%
}
